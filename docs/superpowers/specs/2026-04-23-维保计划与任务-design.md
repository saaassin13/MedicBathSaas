### 5.2 剩余时间计算逻辑

```typescript
/**
 * 计算剩余时间
 * 公式：剩余时间 = (最后一次维保时间 + 维保周期) - 当前时间
 *
 * @param lastMaintenanceTime - 最后一次维保时间（时间戳或Date对象）
 * @param maintenanceCycleHours - 维保周期（小时）
 * @param currentTime - 当前时间（默认使用系统时间）
 * @returns 剩余时间字符串，格式："X天Y小时" 或 "X小时"
 */
function calculateRemainingTime(
  lastMaintenanceTime: Date,
  maintenanceCycleHours: number,
  currentTime: Date = new Date()
): string {
  const nextMaintenanceTime = new Date(
    lastMaintenanceTime.getTime() + maintenanceCycleHours * 60 * 60 * 1000
  );

  const remainingMs = nextMaintenanceTime.getTime() - currentTime.getTime();

  if (remainingMs <= 0) {
    return '已到期';
  }

  const remainingHours = Math.floor(remainingMs / (1000 * 60 * 60));
  const days = Math.floor(remainingHours / 24);
  const hours = remainingHours % 24;

  if (days > 0) {
    return `${days}天${hours}小时`;
  }
  return `${hours}小时`;
}
```

### 5.3 显示规则

| 剩余时间范围 | 显示格式 |
|-------------|---------|
| < 24小时 | `X小时`（红色警示） |
| 24小时 ~ 7天 | `X天Y小时` |
| >= 7天 | `X天Y小时` |
| <= 0 | `已到期`（红色警示） |

## 6. 数据结构

### 6.1 维保计划数据模型

```typescript
interface MaintenancePlan {
  id: string;
  serialNumber: number;        // 序号
  content: string;             // 维保内容描述
  cycleHours: number;          // 维保周期（小时）
  lastMaintenanceTime: string; // 最后一次维保时间，格式：YYYY/MM/DD HH:mm:ss

  remainingTime?: string;      // 计算得出的剩余时间
}

interface MaintenancePlanState {
  plans: MaintenancePlan[];
  selectedHall: string;        // 筛选的奶厅
  loading: boolean;
}
```
